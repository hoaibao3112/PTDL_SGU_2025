import math

# Dy s u vo

# b. Tm s lng cc phn t ca dy s
length = len(a)
print("b. Do dai a: ", length)

# c. In ra gi tr ln nht, gi tr nh nht v gi tr trung bnh
max_val = max(a)
min_val = min(a)
mean_val = sum(a) / len(a)
print("c. Gia tri lon nhat: [{}], gia tri nho nhat: [{}], gia tri trung binh [{:.1f}]".format(max_val, min_val, mean_val))

# d. Tnh phng sai v  lch chun
# Tnh gi tr trung bnh

# Tnh tng bnh phng  lch
sum_squared_deviation = sum((x - mu)**2 for x in a)

# Tnh phng sai ()
variance = sum_squared_deviation / (len(a) - 1)

# Tnh  lch chun (var = )
std_deviation = math.sqrt(variance)

print("d. Phuong sai: [{: .2f}] va do lech chuan: [{: .2f}]".format(variance, std_deviation))

Bi 3. Tnh gn ng s pi dng phng php Monte Carlo

V mt ng trn $C$ c bn knh $r$ ni tip mt hnh vung $C$, v t ngu nhin $N$ chm ln hnh vung. T l cc chm nm trong hnh trn trn tng s chm xp x bng din tch ca hnh trn chia cho hnh vung nh sau: $$ \frac{N_C}{N} \approx \frac{\mathbb{S}_C}{\mathbb{S}_N} $$ ni m $N_C$ l cc chm trong hnh trn, $N$ l tng s chm, $\mathbb{S}_C$ v $\mathbb{S}_N$ ln lt l din tch ca hnh trn v hnh vung. $$ \mathbb{S}_C = \pi * r^2 $$ $$ \mathbb{S}_N = (2*r)^2 = 4r^2 $$ Do , cng thc trn tr thnh: $$ \frac{N_C}{N} \approx \frac{\pi * r^2}{4r^2} = \frac{\pi}{4} $$ Suy ra, ta c gi tr s $\pi$ c tnh nh sau: $$ \pi \approx \frac{4N_C}{N} $$

Cc bn hy lp trnh tnh sp x s $\pi$ dng phng php Monte Carlo. Cho bit cc sai s vi $N = 100$, $N = 10000$, v $N = 1000000$. V th minh ha cho chng trnh.

Hng dn: Vit chng trnh pht sinh $N$ im $(x_i, y_i)$ vi $x_i, y_i \in [-1, -1]$. $(x_i, y_i)$ thuc hnh trn tm $(0, 0)$ bn knh 1 ni tip hnh vung c tm ti $(0, 0)$ di cnh 2 khi v ch khi $x_i^2 + y_i^2 <= 1$.

import numpy as np
import matplotlib.pyplot as plt

# nh ngha cc gi tr N  th nghim
N_values = [100, 1000, 100000]

# Vng lp  tnh ton v hin th kt qu cho tng N
for N in N_values:
    # To cc im ngu nhin trong hnh vung [-1, 1] x [-1, 1]
    x = np.random.uniform(-1, 1, N)
    y = np.random.uniform(-1, 1, N)

    # m s im nm trong ng trn n v (x^2 + y^2 <= 1)
    inside_circle = x**2 + y**2 <= 1
    Nc = np.sum(inside_circle)

    # c lng gi tr 
    pi_approx = 4 * Nc / N
    print(f"N = {N}, Gi tr  c lng: {pi_approx}")

    # V biu  minh ha cho N = 1000
    if N == 1000:
        fig, ax = plt.subplots(figsize=(6, 6))
        circle = plt.Circle((0, 0), 1, color='g', fill=False)
        ax.add_patch(circle)
        ax.plot(x[inside_circle], y[inside_circle], 'b.', label='Trong vng trn')
        ax.plot(x[~inside_circle], y[~inside_circle], 'r.', label='Ngoi vng trn')
        ax.set_xlim(-1, 1)
        ax.set_ylim(-1, 1)
        ax.set_aspect('equal')
        ax.legend()
        plt.title(f'Monte Carlo vi N = {N}')
        plt.savefig(f'pi_N{N}.png')
        plt.show()

# V minh ha ng mi tn
x_arrow = 0.75

import numpy as np

def calc_pi_monte_carlo(n=100):
    pi = 0

    # Sinh ngu nhin n im trong hnh vung [0, 1] x [0, 1]
    x = np.random.rand(n)
    y = np.random.rand(n)

    # m s im nm trong phn t ng trn n v (x^2 + y^2 <= 1)
    inside_circle = x**2 + y**2 <= 1
    num_inside = np.sum(inside_circle)

    # c lng : v y l phn t vng trn, nhn 4  c  y 
    pi = 4 * num_inside / n

    return pi

# Kim tra hm
print(calc_pi_monte_carlo())  # Kt qu vi n=100
print(calc_pi_monte_carlo(1000))  # Kt qu vi n=1000
